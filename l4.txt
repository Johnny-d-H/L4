function getsortedposts() {
  fetch('https://jsonplaceholder.typicode.com/posts')
    .then(response => response.json())
    .then(posts => {
      const sortposts = posts.sort((a, b) => b.title.length - a.title.length);
      console.log('Посты отсортированы по длине тайтла:');
      console.log(sortposts);
    })
    .catch(error => {
      console.log('Ошибка получения постов:', error);
    });
}

function gettodos() {

    .then(response => response.json())

      const untodos = todos.filter(todo => todo.completed === false);

async function getasuncpost() {
  const response = await fetch('https://jsonplaceholder.typicode.com/posts');
  const posts = await response.json();
  const sortedposts = posts.sort((a, b) => b.title.length - a.title.length);
  console.log('Async-посты отсортированы по длине тайтла:');
  console.log(sortedposts);
}
